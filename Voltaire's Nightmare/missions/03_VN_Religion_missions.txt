#Non-Orthodox Christians.
vn_westchristian_missions_1 = {
	slot = 8
	generic = yes
	ai = yes
	potential = {
		religion_group = christian
		NOT = { religion = orthodox }
	}
	
	vn_pious_state = {
		position = 5
		icon = mission_hands_praying
		required_missions = { }
		trigger = {
			if = {
				limit = {
					NOT = {
						OR = {
							religion = catholic
							religion = catholique
						}
					}
				}
				church_power = 200
			}
			else_if = {
				limit = {
					OR = {
						religion = catholic
						religion = catholique
					}
				}
				papal_influence = 200
			}
		}
		effect = {
			add_country_modifier = {
				name = "pious_state_mission"
				duration = 9125
			}
		}
	}
	
	vn_question_of_reform = {
		position = 6
		icon = mission_religious
		required_missions = { }
		trigger = {
			current_age = age_of_reformation
			OR = {
				OR = {
					religion = catholic
					religion = catholique
				}
				OR = {
					religion = protestant
					religion = anabaptist
					religion = anglican
					religion = reformed
				}
			}
		}
		effect = {
			add_country_modifier = {
				name = "question_of_reform_mission"
				duration = 9125
			}
		}
	}

	vn_join_league = {
		position = 7
		icon = mission_european_church
		required_missions = { vn_question_of_reform }
		
		trigger = {
			is_in_league_war = yes
		}
		
		effect = {
			raise_war_taxes = yes
			create_general = {
				tradition = 50
				culture = ROOT
			}
			add_country_modifier = {
				name = "join_league_mission"
				duration = 9125
			}
		}
	}
}
