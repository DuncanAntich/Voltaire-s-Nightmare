	continent = europe
	religion_group = christian
	NOT = { province_id = 2708 }
	is_reformation_center = no
	is_island = no
	NOT = {
		any_neighbor_province = { is_reformation_center = yes }
	}
	NOT = {
		any_province = {
			area = PREV
			is_reformation_center = yes
			religion = $RELIGION$
		}
	}
	NOT = {
		calc_true_if = {
			all_province = {
				superregion = PREV
				is_reformation_center = yes
			}
			amount = 6
		}
	}
}

has_manufactory_trigger = { #Province Scope
	if = {
		limit = {
			OR = {
				trade_goods = naval_supplies
				trade_goods = fish
				trade_goods = salt
				trade_goods = hemp
				trade_goods = wax
			}
		}
		has_building = wharf
	}
	else_if = {
		limit = {
			OR = {
				trade_goods = copper
				trade_goods = iron
				trade_goods = steel
				trade_goods = tin
				trade_goods = plumbum
				trade_goods = munitions
			}
		}
		has_building = weapons
	}
	else_if = {
		limit = {
			OR = {
				trade_goods = wool
				trade_goods = linen
				trade_goods = silk
				trade_goods = carmine
				trade_goods = carpet
			}
		}
		has_building = textile
	}
	else_if = {
		limit = {
			OR = {
				trade_goods = ivory
				trade_goods = slaves
				trade_goods = fur
				trade_goods = honey
				trade_goods = cheese
				trade_goods = gold
			}
		}
		has_building = tradecompany
	}
	else_if = {
		limit = {
			OR = {
				trade_goods = grain
				trade_goods = livestock
				trade_goods = wine
				trade_goods = beer
				trade_goods = horses
			}
		}
		has_building = farm_estate
	}
	else_if = {
		limit = {
			OR = {
				trade_goods = books
				trade_goods = amber
				trade_goods = lumber
				trade_goods = glassware
				trade_goods = chinaware
				trade_goods = marble
			}
		}
		has_building = mills
	}
	else_if = {
		limit = {
			trade_goods = coal
		}
		has_building = furnace
	}
	else = {
		has_building = plantations
	}
}

has_no_manufactory_trigger = { #Province Scope

	NOT = { has_building = tradecompany }
	NOT = { has_building = farm_estate }
	NOT = { has_building = mills }
	NOT = { has_building = plantations }
	NOT = { has_building = furnace }
}
