country_event = {
	id = fbievents.4
	title = "FBIT4"
	desc = "FBID4"
	picture = KING_SICK_IN_BED_eventPicture
	
	fire_only_once = yes
	is_triggered_only = yes
	major = yes

	trigger = {
		normal_or_historical_nations = yes
		current_age = age_of_empire
		NOT = { has_global_flag = lon_split_flag }
		tag = LON
		is_subject = no
		government = monarchy
		religion_group = christian
		OR = {
			any_owned_province = { is_core = CAS }
			any_owned_province = { is_core = GAL }
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}

	immediate = {
		hidden_effect = { set_global_flag = lon_split_flag }
	}

	option = {
		name = "FBIA4"
		ai_chance = { factor = 95 }
		if = {
			limit = { NOT = { exists = CAS } }
			CAS = { define_ruler = { dynasty = ROOT } }
		}
		every_owned_province = {
			limit = { is_core = CAS }
			cede_province = CAS
			remove_core = LON
		}
		if = {
			limit = { NOT = { exists = GAL } }
			GAL = { define_ruler = { dynasty = ROOT } }
		}
		every_owned_province = {
			limit = { is_core = GAL }
			cede_province = GAL
			remove_core = LON
		}

	}
	option = {
		name = "FBIB4"
		ai_chance = { factor = 5 }
		add_corruption = 20
		add_legitimacy = -20
		add_stability = -2
		random_owned_province = {
			limit = { is_core = CAS }
			spawn_rebels = {
				type = nationalist_rebels
				size = 2
			}
		}
		random_owned_province = {
			limit = { is_core = GAL }
			spawn_rebels = {
				type = nationalist_rebels
				size = 2
			}
		}
	}
}
