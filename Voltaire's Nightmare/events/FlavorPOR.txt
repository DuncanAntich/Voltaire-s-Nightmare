# The Meuthen Agreement
country_event = {
	id = flavor_por.3275
	title = "flavor_por.EVTNAME3275"
	desc = "flavor_por.EVTDESC3275"
	picture = DIPLOMACY_eventPicture

	fire_only_once = yes

	trigger = {
		tag = POR
		is_year = 1650
		is_at_war = no
		OR = {
			exists = GBR
			exists = ENG
		}
		iberia_superregion = {
			owned_by = ROOT
			has_port = yes
		}
		NOT = { war_with = ENG }
		NOT = { is_rival = ENG }
		NOT = { war_with = GBR }
		NOT = { is_rival = GBR }
	}

	mean_time_to_happen = {
		months = 100
	}

	
	option = {		# Accept the Agreement
		name = "flavor_por.EVTOPTA3275"
		ai_chance = { factor = 75 }
		if = {
			limit = {
				exists = GBR
			}
			GBR = {
				add_opinion = {
					who = ROOT
					modifier = accepted_meuthen_agreement
				}
			}
			set_country_flag = meuthen_agreement
		}
		if = {
			limit = {
				exists = ENG
				NOT = { has_country_flag = meuthen_agreement }
			}
			ENG = {
				add_opinion = {
					who = ROOT
					modifier = accepted_meuthen_agreement
				}
			}
		}
		random_owned_province = {
			limit = {
				superregion = iberia_superregion
				has_port = yes
			}
			add_base_tax = 1
		}
		clr_country_flag = meuthen_agreement
	}
	option = {		# Denounce the Agreement
		name = "flavor_por.EVTOPTB3275"
		ai_chance = { factor = 25 }
		add_stability_or_adm_power = yes
		if = {
			limit = {
				exists = GBR
			}
			GBR = {
				add_opinion = {
					who = ROOT
					modifier = denounced_meuthen_agreement
				}
			}
			set_country_flag = meuthen_agreement
		}
		if = {
			limit = {
				exists = ENG
				NOT = { has_country_flag = meuthen_agreement }
			}
			ENG = {
				add_opinion = {
					who = ROOT
					modifier = denounced_meuthen_agreement
				}
			}
		}
		clr_country_flag = meuthen_agreement
	}
}

# Christopher Columbus
country_event = {
	id = flavor_por.1
	title = "flavor_por.EVTNAME1"
	desc = "flavor_por.EVTDESC1"
	picture = SHIP_SAILING_eventPicture
	
	
	fire_only_once = yes
	
	trigger = {
		tag = POR
		OR = {
			exploration_ideas = 2
			piracy_ideas = 2
		}
		NOT = { has_global_flag = columbus_happened }
		is_year = 1470
	}

	mean_time_to_happen = {
		months = 1
	}
	
	
	option = {		# Reject his proposal
		name = "flavor_por.EVTOPTA1"
		ai_chance = { factor = 75 }
		set_country_flag = turned_down_columbus
		if = {
			limit = {
				exists = SPA
				SPA = {
					OR = {
						has_idea = quest_for_the_new_world
						has_idea = pir_i2
					}
				}
			}
			SPA = {
				country_event = { id = flavor_spa.1 days = 10 }
			}
		}
		if = {
			limit = {
				NOT = { exists = SPA }
				exists = CAS
				CAS = {
					OR = {
						has_idea = quest_for_the_new_world
						has_idea = pir_i2
					}
				}
			}
			CAS = {
				country_event = { id = flavor_spa.1 days = 10 }
			}
		}
		if = {
			limit = {
				OR = {
					AND = {
						NOT = { exists = SPA }
						NOT = { exists = CAS }
					}
					AND = {
						CAS = { NOT = { has_idea = quest_for_the_new_world } }
						CAS = { NOT = { has_idea = pir_i2 } }
						SPA = { NOT = { has_idea = quest_for_the_new_world } }
						SPA = { NOT = { has_idea = pir_i2 } }
					}
				}
				any_known_country = {
					OR = {
						has_idea = quest_for_the_new_world
						has_idea = pir_i2
					}
					OR = {
						religion = catholic
						religion = catholique
					}
				}
			}
			random_known_country = {
				limit = {
					NOT = { has_country_flag = turned_down_columbus }
					OR = {
						religion = catholic
						religion = catholique
					}
					OR = {
						has_idea = quest_for_the_new_world
						has_idea = pir_i2
					}
				}
				country_event = { id = flavor_spa.1 }
			}
		}
	}
	option = {		# Hire Columbus
		name = "flavor_por.EVTOPTB1"
		ai_chance = { factor = 25 }
		set_global_flag = columbus_happened
		add_years_of_income = -0.10
		define_admiral = {
			name = "Cristóvão Colombo"
			fire = 2
			shock = 2
			manuever = 6
			siege = 0
		}
	}
}
